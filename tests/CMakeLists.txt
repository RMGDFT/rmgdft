MESSAGE(STATUS "")
MESSAGE(STATUS "Creating tests for RMG. The full set requires at least 16Gbytes of memory.")
include(ProcessorCount)
ProcessorCount(NCORES)
if (NOT num_proc)
  if(NOT NCORES EQUAL 0)
      if(NCORES GREATER 24)
          SET(num_proc 16)
      else()
          SET(num_proc ${NCORES})
      endif()
  else()
      SET(num_proc 8)
  endif()
endif()
MESSAGE(STATUS "Tests will use ${num_proc} cores. If this will overload your machine try setting it manually using -Dnum_proc=X.")

SUBDIRS(RMG)
SUBDIRS(RMG_ON)
